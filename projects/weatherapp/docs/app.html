 46class WeatherApp(App):  # type: ignore
 47    """
 48    Compose and run the application.
 49    """
 50
 51    CSS_PATH = "app.tcss"
 52
 53    def compose(self) -> ComposeResult:
 54        """
 55        Compose the application, starting with the input box.
 56        """
 57        input = Input(value="", placeholder="Enter a location...",
 58                      validators=[Validate()], validate_on=["submitted"],
 59                      classes="input")
 60        yield input
 61
 62    def action_report_forecast(self, input: str, feedback: str) -> None:
 63        """
 64        If the feedback gives a failure, report an 'Invalid Input' error.
 65        If the feedback gives a success, report the current forecast
 66        for the given location.
 67        """
 68        if feedback == "success":
 69            location = Location(input)
 70            loc = input
 71            data = Data(loc)
 72
 73            url = data.getURL()
 74            response = requests.get(url)
 75
 76            weatherdata = response.json()
 77
 78            c = data.city(weatherdata)
 79            general_info = data.general(weatherdata)
 80            time_info = data.time(weatherdata)
 81            wind_info = data.wind(weatherdata)
 82
 83            has_numbers = location.has_numbers()
 84
 85            if has_numbers:
 86                location_code = loc.split(",")
 87                z = location_code[0]
 88                b0 = Bar(
 89                    "Current Forecast for: " +
 90                    c +
 91                    " | ZIP Code: [" +
 92                    z + "]",
 93                    classes="top")
 94                b2 = Bar(general_info, classes="general")
 95            else:
 96                b0 = Bar("Current Forecast for: " + input, classes="top")
 97                b2 = Bar(general_info, classes="general")
 98            b1 = Bar(time_info, classes="time")
 99            b3 = Bar(wind_info, classes="air")
100            report = VerticalScroll(b0, Horizontal(b1, b2, b3))
101        else:
102            loc = input
103            invalid = "Invalid Input: Location '" + input + "' not found!"
104            report = VerticalScroll(Bar(invalid, classes="error"))
105
106        self.mount(report)
107        self.call_after_refresh(self.screen.scroll_end, animate=False)
108
109    @on(Input.Submitted)
110    def feedback(self, event: Input.Submitted) -> None:
111        """
112        Based on whether the given input was validated,
113        confirm either a success or a failure.
114        """
115        assert (event.validation_result is not None), (
116            "validation_result is unexpectedly None"
117        )
118        loc = event.input.value
119
120        if event.validation_result.is_valid:
121            self.action_report_forecast(loc, "success")
122        else:
123            self.action_report_forecast(loc, "failure")
124
125    @on(Key)
126    def quit(self, event: Key) -> None:
127        """
128        If the user presses the escape key,
129        close the app.
130        """
131        if event.key == 'escape':
132            exit()
